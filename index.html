1.3自动化造句

Download APK from release page

https://github.com/senzhk/ADBKeyBoard/blob/master/ADBKeyboard.apk 

            
adb install ADBKeyboard.apk 
adb shell ime enable com.android.adbkeyboard/.AdbIME
adb shell ime set com.android.adbkeyboard/.AdbIME   

adb shell am broadcast -a ADB_INPUT_TEXT --es msg '你好嗎? Hello?'
window批处理
            
chcp 65001
@echo off  
setlocal enabledelayedexpansion  

echo 把ADBKeyBoard输入法切换启用并设为默认adb shell settings put secure default_input_method  ime set com.android.adbkeyboard/.AdbIME
TIMEOUT /T 1 /NOBREAK
echo adb shell am start -n com.example.package/.MainActivity
TIMEOUT /T 3 /NOBREAK

:: 设置文件前缀  
set "prefix=output_file_prefix"  

:: 使用for循环遍历以特定前缀开头的文件  
for /f %%f in ('dir /b /a-d /od %prefix%*.txt') do (  
    echo 点击输入框adb shell input tap x y
    TIMEOUT /T 2 /NOBREAK
    echo 正在处理文件: %%f  
    :: 逐行读取文件内容  
    for /f "delims=" %%l in ('type "%%f"') do (  
        echo 输入：%%l  
    echo adb shell input keyevent 66
    )  
    echo 已完成处理文件: %%f  
    TIMEOUT /T 1 /NOBREAK
    echo 点击发送adb shell input tap x y
    TIMEOUT /T 120
    echo ==============================
)  

echo 所有文件处理完毕。  
pause  
endlocal
powershell
            
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8
#设置文件前缀
$prefix = "output_file_prefix"

# 使用Get-ChildItem遍历以特定前缀开头的文件，并按最后修改日期排序
Get-ChildItem -Path "." -Filter "$prefix*.txt" | Sort-Object LastWriteTime | ForEach-Object{
    $file=$_.FullName
    Write-Host "点击输入框（模拟点击）"
    Start-Sleep -Seconds 2
    Write-Host "正在处理文件：$file"

    # 逐行读取文件内容 
    Get-Content -Path $file -Encoding UTF8 | ForEach-Object {
        $line = $_
        Write-Host "输入：${line}"
        #发送adb命令
        Write-Host "adb shell input keyevent 66 (模拟按键，这里仅打印)"  
    }

    Write-Host "已完成处理文件: $file"  
    Start-Sleep -Seconds 1  
    Write-Host "点击发送 (模拟点击操作，这里仅打印)"  
    Read-Host "继续"
    Write-Host "==============================" 
}

1.4文件重命名

window批处理

1.创建文件夹

            
chcp 65001
@echo off  
setlocal  
    
set "inputFile=data.txt"
set "outputFolder=outputFolder"

if not exist "%inputFile%" (
    echo 输入数据文件不存在：%inputFile%
    pause
    exit
)

md %outputFolder%

for /f "delims=" %%a in (%inputFile%) do (
    echo %outputFolder%\%%a
    md %outputFolder%\%%a
)
pause               
endlocal

2.重命名

            
chcp 65001
@echo off
setlocal ENABLEDELAYEDEXPANSION

REM 定义文件夹的入口
set "input=folderdemo"
set "numberclass=0"
REM 定义命名分类
set "arr[0]=原图"
set "arr[1]=水彩"
set "arr[2]=赛博"
set "arr[3]=卡通"

REM 读取folderdemo文件夹下的子目录
for /d %%a in (%input%\*) do (
    echo %%a
    
    REM 读取子目录文件下的文件名，文件名按时间顺序排列
    FOR /f "delims=" %%b IN ('dir /b /a-d /od %%a\*') DO (		
        echo %%b对应的序号是!numberclass!

        if !numberclass! equ 0 (
            ren "%%a\%%b" "%%~na meitu !arr[0]!.png"
            echo %%a\%%b命名为：%%~na meitu !arr[0]!.png
        )
        if !numberclass! equ 1 (
            ren "%%a\%%b" "%%~na meitu !arr[1]!.png"
            echo %%a\%%b命名为：%%~na meitu !arr[1]!.png
        )
        if !numberclass! equ 2 (
            ren "%%a\%%b" "%%~na meitu !arr[2]!.png"
            echo %%a\%%b命名为：%%~na meitu !arr[2]!.png
        )
        if !numberclass! equ 3 (
            ren "%%a\%%b" "%%~na meitu !arr[3]!.png"
            echo %%a\%%b命名为：%%~na meitu !arr[3]!.png
        )
        set /a "numberclass+=1"
        if !numberclass! gtr 3 (set "numberclass=0")
        
    )
)

pause
endlocal

3.移动文件

            
chcp 65001
@echo off
SETLOCAL ENABLEDELAYEDEXPANSION

REM 定义用户输入变量
SET /P result="请输入文件夹："
pause

REM 定义要搜索的文件夹
set "target=outputdir"

REM 搜索目标文件夹下的所有文件，不包括目录
for /f %%a in ('dir /b /a-d /s %target%\*') do (
    echo %%a
    REM 将这些图片复制到一个文件
    copy "%%a" "%result%\"
)

pause
endlocal
powershell

1.创建文件夹

            
# 检查当前文件夹中是否存在名为data.txt的文件 
$dataFile = Get-ChildItem -Path . -Filter data.txt -File

if ($dataFile){
    # 如果不存在，逐行读取文件的内容
    foreach ($line in Get-Content $dataFile -Encoding UTF8){
        # 在指定路径下创建一个名为MyNewFolder的文件夹  
        New-Item -ItemType Directory -Path .\outputdir\$line
    }
} else {
    # 如果不存在，输出没有然后退出程序  
    Write-Host "当前文件夹下没有数据文件data.txt"
    #exit 1
}

2.重命名

            
$rootDirPath = "outputdir"
$categories = @("水果", "蔬菜", "肉类", "零食")  

# 只遍历根目录下的直接子目录 
Get-ChildItem -Path $rootDirPath -Directory | ForEach-Object {
    $subDir = $_
    $counter = 0
    # 遍历每个子目录中的文件，按名称排序（默认按名称升序）
    Get-ChildItem -Path $subDir.FullName -File | Sort-Object Nmae | ForEach-Object {
        $file = $_
        #构建新文件名
        $newName = "{0} {1}.txt" -f $subDir.Name, $categories[$counter % $categories.Length]
        #重命名文件
        Rename-Item -Path $file.FullName -NewName $newName
        # 输出重命名信息  
        Write-Host "将$($file.FullName) 命名为：$newName"  
        #更新计数器
        $counter++       
    }
}

3.移动文件

            
#请求用户输入目标文件夹路径
$result = Read-Host -Prompt "请输入文件夹（用于存放复制的文件）"

#定义要搜索的源文件
$target = "outputdir"

#检查输入文件夹是否存在，如果不存在则创建它
if (!(Test-Path -Path $result -PathType Container)){
    Write-Host "目标文件夹不存在，正在创建..."
    New-Item -ItemType Directory -Path $result | Out-Null
}

# 搜索目标文件夹下的所有文件（不包括目录），并将它们复制到用户指定的文件夹 
Get-ChildItem -Path $target -Recurse -File | ForEach-Object {
    $sourcePath = $_.FullName
    $destPath = Join-Path -Path $result -ChildPath $_.Name

    Write-Host "正在复制：$sourcePath 到 $destPath" 
    Copy-Item -Path $sourcePath -Destination $destPath
}

1.5刷视频

window批处理
            
chcp 65001
@echo off
SETLOCAL ENABLEDELAYEDEXPANSION

SET /P input="输入间隔时间："
echo %input%

FOR /L %%a IN (1,1,10000) DO (
    TIMEOUT /T %input%
    adb shell input swipe 300 600 300 100
    echo %%a
)
